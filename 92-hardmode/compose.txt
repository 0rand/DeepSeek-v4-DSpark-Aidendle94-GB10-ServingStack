services:
  ds4:
    image: ${IMAGE:?set IMAGE in .env}
    container_name: ${NAME:-ds4-dspark}
    network_mode: host
    ipc: host
    shm_size: 32g
    gpus: all
    init: true
    ulimits:
      memlock: -1
      nofile: 1048576
      stack: 67108864
    volumes:
      - ${HF_CACHE:?set HF_CACHE in .env}:/hf_cache
      - ${VLLM_CACHE:?set VLLM_CACHE in .env}:/cache
      - ${TILELANG_CACHE:?set TILELANG_CACHE in .env}:/root/.tilelang
    devices:
      - /dev/infiniband:/dev/infiniband
    environment:
      # ---- GPU ----
      CUDA_VISIBLE_DEVICES: "0"
      CUDA_DEVICE_ORDER: PCI_BUS_ID
      CUTE_DSL_ARCH: sm_121a
      GPU_MEMORY_UTILIZATION: ${GPU_MEMORY_UTILIZATION:-0.8}
      # ---- RoCE networking ----
      NCCL_NET: IB
      NCCL_IB_DISABLE: "0"
      NCCL_IB_HCA: ${NCCL_IB_HCA:?your RoCE HCAs}
      NCCL_SOCKET_IFNAME: ${NCCL_SOCKET_IFNAME:?your RoCE netdevs}
      GLOO_SOCKET_IFNAME: ${CONTROL_IF:?control netdev}
      TP_SOCKET_IFNAME: ${CONTROL_IF}
      MN_IF_NAME: ${CONTROL_IF}
      OMPI_MCA_btl_tcp_if_include: ${CONTROL_IF}
      NCCL_IB_GID_INDEX: ${NCCL_IB_GID_INDEX:-5}
      NCCL_CROSS_NIC: "1"
      NCCL_DEBUG: Info
      NCCL_IB_TIMEOUT: "24"
      NCCL_P2P_LEVEL: "2"
      
      NCCL_IB_QPS_PER_CONNECTION: "8"
      NCCL_BUFFERSIZE: "16777216"
      
      NCCL_SOCKET_FAMILY: "inet"
      NCCL_PROTO: LL,LL128,Simple
      
      NCCL_CUMEM_ENABLE: "0"
      NCCL_IGNORE_CPU_AFFINITY: "1"
      
     # ---- runtime ----
      HF_HUB_OFFLINE: "1"
      PYTORCH_CUDA_ALLOC_CONF: expandable_segments:True
      VLLM_PREFIX_CACHE_RETENTION_INTERVAL: "4096"
      VLLM_USE_AOT_COMPILE: "1"
      VLLM_USE_MEGA_AOT_ARTIFACT: "0"
      VLLM_USE_BREAKABLE_CUDAGRAPH: "0"
      VLLM_USE_V2_MODEL_RUNNER: "1"
      VLLM_USE_FLASHINFER_SAMPLER: "1"
      VLLM_USE_B12X_MOE: "1"
      VLLM_DSPARK_REPLICATE_MARKOV_W1: "1"
      ASYNC_SCHED: ${ASYNC_SCHED:-1}
      # ---- model / serve ----
      MODEL_PATH: ${MODEL_PATH:-deepseek-ai/DeepSeek-V4-Flash-DSpark}
      PORT: ${PORT:-8000}
      GPU_MEM: ${GPU_MEMORY_UTILIZATION:-0.8}
      SPEC_TOKENS: ${SPEC_TOKENS:-3}
      TP_SIZE: ${TP_SIZE:-2}
      MAX_NUM_SEQS: ${MAX_NUM_SEQS:-128}
      MAX_NUM_BATCHED_TOKENS: ${MAX_NUM_BATCHED_TOKENS:-8192}
      GRAPH_CAP: ${GRAPH_CAP:-128}
      # ---- per-node ----
      NODE_RANK: "${NODE_RANK:?0 on head, 1 on worker}"
      HEADLESS: "${HEADLESS:-}"
      MASTER_ADDR: "${MASTER_ADDR:?head-node RoCE IP}"
    command:
      - /bin/bash
      - -lc
      - |
        set -euo pipefail
        unset NCCL_GRAPH_FILE NCCL_GRAPH_DUMP_FILE VLLM_B12X_MLA_EXTEND_MAX_CHUNKS
        export VLLM_ENABLE_PCIE_ALLREDUCE=0 VLLM_PCIE_ALLREDUCE_BACKEND=cpp
        # Auto-detect the RoCE-v2 IPv4 GID index (it re-numbers across reboots).
        for HCA in $(echo "$${NCCL_IB_HCA}" | tr ',' ' '); do
          for i in $(seq 0 15); do
            t=$(cat /sys/class/infiniband/$$HCA/ports/1/gid_attrs/types/$$i 2>/dev/null || true)
            g=$(cat /sys/class/infiniband/$$HCA/ports/1/gids/$$i 2>/dev/null || true)
            case "$$t" in *"RoCE v2"*) case "$$g" in *0000:0000:0000:0000:0000:ffff:*) export NCCL_IB_GID_INDEX=$$i; break 2;; esac;; esac
          done
        done
        echo "[dspark] NODE_RANK=$${NODE_RANK} SPEC=dspark/$${SPEC_TOKENS} GID=$${NCCL_IB_GID_INDEX}"
        if [ "$${ASYNC_SCHED:-1}" = "1" ]; then ASYNC_ARG="--async-scheduling"; else ASYNC_ARG=""; fi
        exec vllm serve "$${MODEL_PATH}" \
          --served-model-name ${MODEL_NAME:-ChatGPTN} \
          --host 0.0.0.0 --port "$${PORT}" --trust-remote-code \
          --kv-cache-dtype fp8 --block-size 256 --load-format auto \
          --tensor-parallel-size "$${TP_SIZE}" \
          --gpu-memory-utilization "$${GPU_MEM}" \
          --max-model-len 393216 --max-num-seqs "$${MAX_NUM_SEQS}" \
          --max-num-batched-tokens "$${MAX_NUM_BATCHED_TOKENS}" \
          --max-cudagraph-capture-size "$${GRAPH_CAP}" \
          --compilation-config '{"cudagraph_mode":"FULL_AND_PIECEWISE","custom_ops":["all"]}' \
          $${ASYNC_ARG} --no-scheduler-reserve-full-isl \
          --enable-chunked-prefill --enable-prefix-caching --enable-flashinfer-autotune \
          --tokenizer-mode deepseek_v4 --tool-call-parser deepseek_v4 --reasoning-parser deepseek_v4 \
          --enable-auto-tool-choice \
          --generation-config vllm \
          --default-chat-template-kwargs.temperature=0.48 \
          --default-chat-template-kwargs.thinking=true --default-chat-template-kwargs.reasoning_effort=high \
          --attention-backend FLASHINFER_MLA_SPARSE_DSV4 --moe-backend b12x \
          --disable-custom-all-reduce \
          --nnodes 2 --node-rank "$${NODE_RANK}" --master-addr "$${MASTER_ADDR}" --master-port 25000 \
          $${HEADLESS:+--headless} \
          --speculative-config "{\"method\":\"dspark\",\"num_speculative_tokens\":$${SPEC_TOKENS},\"draft_sample_method\":\"probabilistic\"}"
